function numberOfPairs(gloves){
  let sum = 0;
  let colors = new Set(gloves);
  for (let color of colors){
    let amount = gloves.filter(x => x === color).length;
    if (amount >= 2) {
      sum += Math.floor(amount / 2)
    }
  }
  return sum
}
